CREATE TABLE appUsers (
    userId INT PRIMARY KEY IDENTITY(1,1),
    userName NVARCHAR(100) NOT NULL,
    email NVARCHAR(100) UNIQUE NOT NULL,
    passwordHash NVARCHAR(255) NOT NULL,
    role NVARCHAR(50) DEFAULT 'Employee',
    createdAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE Attendance (
    attendanceId INT PRIMARY KEY IDENTITY(1,1),
    userId INT NOT NULL FOREIGN KEY REFERENCES appUsers(userId),
    checkInTime DATETIME NULL,
    checkOutTime DATETIME NULL,
    status AS 
        CASE 
			WHEN checkInTime IS NULL AND checkOutTime IS NULL THEN 'On Leave'
			WHEN checkOutTime IS NULL THEN 'Active'
			ELSE 'Not Active'
		END PERSISTED,
    sessionDuration AS DATEDIFF(MINUTE, checkInTime, checkOutTime) PERSISTED,
    createdAt DATETIME DEFAULT GETDATE(),
    attendanceDate DATE DEFAULT GETDATE(),
    onLeave BIT DEFAULT 0
);

CREATE TABLE locationLogs (
    locationId INT PRIMARY KEY IDENTITY(1,1),
    attendanceId INT NOT NULL FOREIGN KEY REFERENCES attendance(attendanceId),
    latitude FLOAT NOT NULL,
    longitude FLOAT NOT NULL,
    loggedAt DATETIME DEFAULT GETDATE()
);

CREATE TABLE leaveBalance (
    balanceId INT PRIMARY KEY IDENTITY(1,1),
    userId INT NOT NULL FOREIGN KEY REFERENCES appUsers(userId),
    year INT NOT NULL, -- The calendar year for the leave balance
    totalLeaves INT DEFAULT 12, -- Default annual allotment
    usedLeaves INT DEFAULT 0, -- Leaves already used
    remainingLeaves AS (totalLeaves - usedLeaves) PERSISTED, -- Automatically calculated
    lastUpdated DATETIME DEFAULT GETDATE()
);

    CREATE TABLE LeaveRequests (
        LeaveId INT PRIMARY KEY IDENTITY(1,1),
        UserId INT NOT NULL FOREIGN KEY REFERENCES appUsers(UserId),
        LeaveType NVARCHAR(50) NOT NULL, -- e.g., Sick Leave, Casual Leave
        StartDate DATE NOT NULL,
        EndDate DATE NOT NULL,
        Status NVARCHAR(50) DEFAULT 'Pending', -- Pending, Approved, Rejected
        Reason NVARCHAR(255) NULL,
        RequestedAt DATETIME DEFAULT GETDATE(),
        ApprovedBy INT NULL FOREIGN KEY REFERENCES appUsers(UserId),
        ApprovedAt DATETIME NULL,
        RejectedBy INT NULL FOREIGN KEY REFERENCES appUsers(UserId),
        RejectedAt DATETIME NULL,
        CONSTRAINT CHECK_EndDate CHECK (EndDate >= StartDate)
    );

Create TRIGGER createLeaveBalance
ON appUsers
AFTER INSERT
AS
BEGIN
    INSERT INTO leaveBalance (userId, year, totalLeaves)
    SELECT userId, YEAR(GETDATE()), 12
    FROM inserted;
END;



Create PROCEDURE approveLeave
    @LeaveId INT,
    @actionId INT,
    @applicantUserId INT,
	@action VARCHAR(50)
AS
BEGIN
	IF @action = 'Approved'
		BEGIN
			BEGIN TRANSACTION;
				UPDATE LeaveRequests
				SET Status = 'Approved', ApprovedBy =  @actionId, ApprovedAt = GETDATE()
				WHERE LeaveId = @LeaveId;

				IF @@ROWCOUNT = 0
				BEGIN
					ROLLBACK TRANSACTION;
					RAISERROR('LeaveId does not exist.', 16, 1); 
					RETURN; 
				END

				UPDATE LeaveBalance
				SET UsedLeaves = UsedLeaves + 1
				WHERE UserId = @applicantUserId;

				IF @@ROWCOUNT = 0
				BEGIN
					ROLLBACK TRANSACTION;
					RAISERROR('UserId does not exist or no leave balance to update.', 16, 1);
					RETURN;
				END
			COMMIT TRANSACTION;
	    END
	ELSE IF @action = 'Rejected'
	BEGIN
		BEGIN TRANSACTION;
			update LeaveRequests set status = 'Rejected', RejectedBy =  @actionId , RejectedAt = GETDATE() where LeaveId = @LeaveId;
			IF @@ROWCOUNT = 0
				BEGIN
					ROLLBACK TRANSACTION;
					RAISERROR('LeaveId does not exist.', 16, 1); 
					RETURN; 
				END
		COMMIT TRANSACTION;
	END
	ELSE
    BEGIN
        RAISERROR('Invalid action provided.', 16, 1);
        RETURN;
    END
END;


CREATE PROCEDURE approveLeaveUpdated
    @LeaveId INT,
    @actionId INT,
    @applicantUserId INT,
    @action VARCHAR(50)
AS
BEGIN
    IF @action = 'Approved'
    BEGIN
        BEGIN TRANSACTION;
            -- Approve the leave request
            UPDATE LeaveRequests
            SET Status = 'Approved', ApprovedBy = @actionId, ApprovedAt = GETDATE()
            WHERE LeaveId = @LeaveId;

            IF @@ROWCOUNT = 0
            BEGIN
                ROLLBACK TRANSACTION;
                RAISERROR('LeaveId does not exist.', 16, 1);
                RETURN;
            END

            -- Update the leave balance
            --UPDATE LeaveBalance
            --SET UsedLeaves = UsedLeaves + 1
            --WHERE UserId = @applicantUserId;
			DECLARE @DateDif INT;
			SELECT @DateDif = DATEDIFF(DAY, StartDate, EndDate) + 1 from  LeaveRequests where LeaveId = @LeaveId;

			UPDATE LeaveBalance
            SET UsedLeaves = UsedLeaves + @DateDif
            WHERE UserId = @applicantUserId;

            IF @@ROWCOUNT = 0
            BEGIN
                ROLLBACK TRANSACTION;
                RAISERROR('UserId does not exist or no leave balance to update.', 16, 1);
                RETURN;
            END

            -- Fetch leave period from LeaveRequests
            DECLARE @StartDate DATE, @EndDate DATE;
            SELECT @StartDate = StartDate, @EndDate = EndDate
            FROM LeaveRequests
            WHERE LeaveId = @LeaveId;

            -- Insert leave entries into Attendance table ----->
			BEGIN TRY
				 WHILE @StartDate <= @EndDate
				 BEGIN
					INSERT INTO Attendance (userId, checkInTime, checkOutTime, attendanceDate, onLeave)
					VALUES (@applicantUserId, NULL, NULL, @StartDate, 1);

					SET @StartDate = DATEADD(DAY, 1, @StartDate);
				 END	
			END TRY
			BEGIN CATCH
				ROLLBACK TRANSACTION
				RAISERROR('Error in inserting leave entries', 16, 1);
                RETURN;

			END CATCH
			-- Insert leave entries into Attendance table  <----

        COMMIT TRANSACTION;
    END
    ELSE IF @action = 'Rejected'
    BEGIN
        BEGIN TRANSACTION;
            -- Reject the leave request
            UPDATE LeaveRequests
            SET Status = 'Rejected', RejectedBy = @actionId, RejectedAt = GETDATE()
            WHERE LeaveId = @LeaveId;

            IF @@ROWCOUNT = 0
            BEGIN
                ROLLBACK TRANSACTION;
                RAISERROR('LeaveId does not exist.', 16, 1);
                RETURN;
            END
        COMMIT TRANSACTION;
    END
    ELSE
    BEGIN
        RAISERROR('Invalid action provided.', 16, 1);
        RETURN;
    END
END;
